Quick Start

Load the library and create your first window in seconds. This is the recommended starting point for any script.

local UI = loadstring(game:HttpGet("https://xan.bar/init.lua"))()

local Window = UI.New({
    Title = "My Script",
    Theme = "Default",
    Size = UDim2.new(0, 580, 0, 420),
    ShowUserInfo = true,
    ShowActiveList = true
})

local Main = Window:AddTab("Main", UI.Icons.Home)

Main:AddSection("Features")
Main:AddToggle("Enable Feature", { Default = true }, function(v)
    print("Toggled:", v)
end)

Main:AddSlider("Walk Speed", { Min = 16, Max = 100, Default = 16 }, function(v)
    local char = game.Players.LocalPlayer.Character
    if char and char:FindFirstChild("Humanoid") then
        char.Humanoid.WalkSpeed = v
    end
end)

Main:AddButton("Click Me", function()
    UI.Success("Button Clicked!", "It works!")
end)

    Config.Enabled = v
end)

combat:AddSlider("FOV Size", { Min = 50, Max = 400, Default = 180, Flag = "aim_fov" }, function(v)
    Config.FOV = v
end)

combat:AddSlider("Smoothing", { Min = 0.05, Max = 1, Default = 0.15, Increment = 0.01, Flag = "aim_smooth" }, function(v)
    Config.Smooth = v
end)

combat:AddDropdown("Hitbox", {"Head", "Torso", "Random"}, function(v)
    Config.Hitbox = v
end)

combat:AddKeybind("Toggle Key", { Default = Enum.KeyCode.X, Flag = "aim_key" }, function()
    Config.Enabled = not Config.Enabled
    UI.Pill(Config.Enabled and "Aimbot ON" or "Aimbot OFF")
end)

-- Add FOV circle
local fov = UI.FOV({ Radius = 180, FollowMouse = true })
UI.OnFlag("aim_fov", function(v) fov:SetRadius(v) end)

UI.Success("Loaded!", "Press X to toggle aimbot")

A full commercial-grade hub with authentication, game selection, and branding:

--[[
    PRODUCTION HUB TEMPLATE
    Copy and customize this for your own hub
]]

local UI = loadstring(game:HttpGet("https://xan.bar/init.lua"))()

-- ═══════════════════════════════════════
-- CONFIGURATION (Edit these!)
-- ═══════════════════════════════════════
local CONFIG = {
    HubName = "My Hub",
    Version = "1.0.0",
    Logo = UI.Logos.XanBar,           -- Replace with your logo
    Theme = "Default",                -- Default, Rose, Midnight, etc.
    DiscordInvite = "discord.gg/xxx",
    ApiUrl = "https://yoursite.com/api",
}

-- ═══════════════════════════════════════
-- LICENSE VERIFICATION
-- ═══════════════════════════════════════
local UserData = nil
local HttpService = game:GetService("HttpService")

local function VerifyLicense(key)
    local success, result = pcall(function()
        return game:HttpGet(
            CONFIG.ApiUrl .. "/verify?key=" .. HttpService:UrlEncode(key)
        )
    end)
    
    if not success then
        return false, "Connection failed"
    end
    
    local data = HttpService:JSONDecode(result)
    if data.success then
        UserData = data
        return true
    end
    return false, data.message or "Invalid key"
end

-- ═══════════════════════════════════════
-- SPLASH SCREEN
-- ═══════════════════════════════════════
UI.Splash({
    Title = CONFIG.HubName,
    Subtitle = "v" .. CONFIG.Version,
    Logo = CONFIG.Logo,
    Duration = 2,
    OnComplete = function()
        -- Show login after splash
        UI.Login({
            Title = CONFIG.HubName,
            Subtitle = "Enter your license key",
            Logo = CONFIG.Logo,
            ShowSignup = false,
            ShowForgotPassword = false,
            OnLogin = function(_, key)
                local success, msg = VerifyLicense(key)
                if success then
                    CreateMainWindow()
                end
                return success, msg
            end
        })
    end
})

-- MAIN WINDOW

Layouts

Two layout styles are available. Choose based on your script complexity and tab count.

Default (Sidebar)

Vertical sidebar with tabs on the left. Best for scripts with 4+ tabs. Shows tab names and icons.

Traditional (Top Tabs)

Login with Node.js Backend

If you prefer Node.js over PHP:

// server.js - Express API
const express = require('express');
const cors = require('cors');
const mysql = require('mysql2/promise');

const app = express();
app.use(cors());

const pool = mysql.createPool({
    host: 'localhost',
    user: 'root',
    password: 'password',
    database: 'myhub'
});

app.get('/api/verify', async (req, res) => {
    const { key, hwid } = req.query;
    
    try {
        const [rows] = await pool.query(
            'SELECT * FROM licenses WHERE license_key = ? AND expires > NOW()',
            [key]
        );
        
        if (rows.length === 0) {
            return res.json({ success: false, message: 'Invalid or expired key' });
        }
        
        const license = rows[0];
        
        if (license.hwid && license.hwid !== hwid) {
            return res.json({ success: false, message: 'HWID mismatch' });
        }
        
        if (!license.hwid && hwid) {
            await pool.query('UPDATE licenses SET hwid = ? WHERE id = ?', [hwid, license.id]);
        }

local HttpService = game:GetService("HttpService")
local GameIconCache = {}

local function GetGameIcon(gameName, callback)
    local cacheKey = gameName:lower():gsub("[^a-z0-9]+", "_")
    
    -- Check cache first
    if GameIconCache[cacheKey] then
        callback(GameIconCache[cacheKey])
        return
    end
    
    local url = "https://api.xan.bar/api/games/lookup?name=" .. HttpService:UrlEncode(gameName)
    
    task.spawn(function()
        local success, response = pcall(function()
            return request({ Url = url, Method = "GET" })
        end)
        
        if success and response.StatusCode == 200 then
            local data = HttpService:JSONDecode(response.Body)
            if data.success and data.game then
                local icon = data.game.backup_rbxasset or data.game.rbxthumb
                GameIconCache[cacheKey] = icon
                callback(icon)
                return
            end
        end
        callback(nil)
    end)
end

Building Your Own Plugin

A well-built plugin should:

  • Sync with Xan's current theme
  • React to theme changes
  • Clean up properly on unload
  • Store connections for cleanup

Plugin Template

local Xan = getgenv().XanBarInstance
if not Xan then
    warn("Xan UI not loaded. Load xan.bar first!")
    return
end

local MyPlugin = {
    Name = "MyPlugin",
    Version = "1.0.0",
    _Connections = {},
    _Instances = {},
    _Loaded = false
}

-- Access theme colors (always use Xan.CurrentTheme, not cached values)
function MyPlugin:GetColors()
    return {
        Accent = Xan.CurrentTheme.Accent,
        Background = Xan.CurrentTheme.Background,
        Card = Xan.CurrentTheme.Card,
        Text = Xan.CurrentTheme.Text,
        TextSecondary = Xan.CurrentTheme.TextSecondary,
        Border = Xan.CurrentTheme.CardBorder
    }
end

-- Subscribe to theme changes
function MyPlugin:SetupThemeSync()
    -- Xan fires this when theme changes
    if Xan.OnThemeChanged then
        local conn = Xan:OnThemeChanged(function(newTheme)
            self:ApplyTheme(newTheme)
        end)
        table.insert(self._Connections, conn)
    end
end

-- Apply theme to your UI elements
function MyPlugin:ApplyTheme(theme)
    theme = theme or Xan.CurrentTheme
    
    for _, instance in pairs(self._Instances) do
        if instance:IsA("Frame") then
            instance.BackgroundColor3 = theme.Card
        elseif instance:IsA("TextLabel") then
            instance.TextColor3 = theme.Text
        end
    end
end

-- Clean up everything
function MyPlugin:Unload()
    -- Disconnect all connections
    for _, conn in pairs(self._Connections) do
        pcall(function() conn:Disconnect() end)
    end
    self._Connections = {}
    
    -- Destroy all instances
    for _, instance in pairs(self._Instances) do
        pcall(function() instance:Destroy() end)
    end
    self._Instances = {}
    
    self._Loaded = false
end

-- Register with Xan's unload system
function MyPlugin:Register()
    if Xan.RegisterPlugin then
        Xan:RegisterPlugin(self.Name, self)
    end
    
    -- Also hook into Xan's global unload

                checkConn:Disconnect()
            end
        end)

    end
end

Theme Sync

Making your plugin respond to theme changes ensures a seamless user experience when users switch themes.

The Problem

-- BAD: Colors captured at load time, won't update
local accent = Xan.CurrentTheme.Accent  -- Captured once!

function CreateButton()
    button.BackgroundColor3 = accent  -- Always the old color
end

The Solution

-- GOOD: Always read from CurrentTheme
function CreateButton()
    button.BackgroundColor3 = Xan.CurrentTheme.Accent  -- Always current!
end

-- BEST: Subscribe to theme changes
local MyPlugin = { _Elements = {} }

function MyPlugin:OnThemeChanged(newTheme)
    for _, element in pairs(self._Elements) do
        if element.Type == "AccentFrame" then
            element.Instance.BackgroundColor3 = newTheme.Accent
        elseif element.Type == "CardFrame" then
            element.Instance.BackgroundColor3 = newTheme.Card
        elseif element.Type == "TextLabel" then
            element.Instance.TextColor3 = newTheme.Text
        end
    end
end

-- Track elements for theming
function MyPlugin:AddElement(instance, elementType)
    table.insert(self._Elements, {
        Instance = instance,
        Type = elementType
    })
end

-- Register the callback
if Xan.OnThemeChanged then
    Xan:OnThemeChanged(function(theme)
        MyPlugin:OnThemeChanged(theme)
    end)
end

Plugin Lifecycle

A robust plugin handles its entire lifecycle: initialization, running, and cleanup.

Lifecycle Stages

1. Load

Check if Xan exists, initialize state, don't create UI yet.

2. Create

Build UI, store all instances and connections for cleanup.

3. Run

Plugin is active, handle events, sync with theme changes.

4. Unload

Disconnect events, destroy instances, clear references.

Unload Pattern

local Plugin = {
    _Connections = {},  -- Store all RBXScriptConnections
    _Instances = {},    -- Store all created Instances
    _Active = false
}

function Plugin:Connect(signal, callback)
    local conn = signal:Connect(callback)
    table.insert(self._Connections, conn)
    return conn
end

function Plugin:Create(className, properties)
    local instance = Instance.new(className)
    for k, v in pairs(properties) do
        instance[k] = v
    end
    table.insert(self._Instances, instance)
    return instance
end

function Plugin:Unload()
    if not self._Active then return end
    self._Active = false
    
    -- Disconnect all
    for i = #self._Connections, 1, -1 do
        pcall(function()
            self._Connections[i]:Disconnect()
        end)
        self._Connections[i] = nil
    end
    
    -- Destroy all
    for i = #self._Instances, 1, -1 do
        pcall(function()
            self._Instances[i]:Destroy()
        end)
        self._Instances[i] = nil
    end
end

-- Auto-unload when Xan unloads
function Plugin:WatchForXanUnload()
    self:Connect(game:GetService("RunService").Heartbeat, function()
        if not getgenv().XanBarInstance then
            self:Unload()
        end
    end)
end

Best Practice: Always use wrapper functions like Plugin:Connect() and Plugin:Create() instead of raw Instance.new() and signal:Connect(). This ensures you never forget to track something for cleanup.

Critical: Forward Declarations

When adding new features that reference each other, you MUST use forward declarations to avoid nil errors.

-- BAD: Will cause "attempt to index nil" error
for i, item in ipairs(items) do
    btn.Click:Connect(function()
        newElement.Visible = true  -- ERROR: newElement doesn't exist yet!
    end)
end
local newElement = CreateElement()

-- GOOD: Forward declare before, assign after
local newElement = nil  -- Forward declaration

for i, item in ipairs(items) do
    btn.Click:Connect(function()
        if newElement then  -- Safe check
            newElement.Visible = true
        end
    end)
end

newElement = CreateElement()  -- Assign after loop

Stitching & Building

After editing modules, you must stitch them back into a single file for deployment.

Using xanbar_splitter.py

The splitter tool handles both splitting and stitching. Located at tools/xanbar_splitter.py.

# Navigate to project root
cd blackgf

# Stitch modules into init_stitched.lua
python tools/xanbar_splitter.py stitch

# Split init.lua into modules (rebuilds line ranges)
python tools/xanbar_splitter.py split

What the Tool Does

  • stitch - Reads all numbered .lua files in modules/ in order, concatenates them, outputs to init_stitched.lua
  • split - Reads init.lua and creates individual module files based on line definitions

Changing Line Definitions

The line ranges for each module are defined inside xanbar_splitter.py. To change where modules start/end, edit the add_module() calls:

# In xanbar_splitter.py, find the split_init() function:

add_module('00_header', 1, 36, 'Header comments')
add_module('01_services', 37, 91, 'Services + ZIndex')
add_module('02_xan_core', 92, 128, 'Xan table + forward decl')
# ... more modules ...

# Format: add_module(name, start_line, end_line, description)
# Lines are 1-indexed and inclusive

After changing line definitions, run python tools/xanbar_splitter.py split to regenerate the module files with the new ranges.
